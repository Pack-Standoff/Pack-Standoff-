// Função para abrir a galeria de imagens e salvar a URL da imagem selecionada function abrirAbaImagem() { var inputImagem = document.createElement("input"); inputImagem.type = "file"; inputImagem.accept = "image/*"; inputImagem.style.display = "none"; document.body.appendChild(inputImagem); inputImagem.addEventListener("change", function() { var imagemSelecionada = inputImagem.files[0]; if (imagemSelecionada) { var imagemURL = URL.createObjectURL(imagemSelecionada); // Exiba a imagem escolhida acima do botão var botaoImagem = document.getElementById("botao-imagem"); botaoImagem.style.backgroundImage = `url('${imagemURL}')`; botaoImagem.style.backgroundSize = "cover"; // Salve a URL da imagem no armazenamento local salvarURLImagem(imagemURL); } inputImagem.remove(); }); } // Função para salvar a URL da imagem no armazenamento local function salvarURLImagem(url) { localStorage.setItem('urlImagem', url); } // Carregue a URL da imagem do armazenamento local e defina-a como imagem de fundo do botão var urlSalva = localStorage.getItem('urlImagem'); if (urlSalva) { var botaoImagem = document.getElementById("botao-imagem"); botaoImagem.style.backgroundImage = `url('${urlSalva}')`; botaoImagem.style.backgroundSize = "cover"; }
